# SubScope MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Next.js app where users input a Substack handle and see profile stats + top 10 posts sorted by heart reactions.

**Architecture:** A single-page React UI posts the handle to a Next.js API route (`/api/profile`) which fetches the Substack `@handle` page server-side, parses `window._preloads` to extract profile data + user ID, then calls the Substack posts API. The client renders profile card + sorted post list. All external Substack fetches live in the API route to avoid CORS.

**Tech Stack:** Next.js 15 (App Router), TypeScript, Tailwind CSS v4, Node.js built-in `fetch`

## Confirmed Data Shapes (verified against live API)

**Profile** (from `window._preloads` → `.profile`):
- `id` — number (user ID for posts API)
- `name` — string
- `handle` — string
- `photo_url` — string
- `bio` — string
- `subscriberCount` — string (e.g. "1.2K") or null
- `subscriberCountNumber` — number or null
- `followerCount` — number
- `primaryPublication.name` — string (publication name)
- `primaryPublication.paid_subscription_benefit_title` — indicates paid tier

**Posts** (from `/api/v1/profile/posts`):
- `posts[]` — array of post objects
  - `title` — string
  - `subtitle` — string or null
  - `canonical_url` — string
  - `reactions['❤']` — number (heart count)
  - `restacks` — number
  - `audience` — `'only_paid'` | `'everyone'` (free vs paid)
  - `post_date` — ISO date string

## Task 3: Build the API Route

**Files:**
- Create: `src/app/api/profile/route.ts`

This route:
1. Accepts `GET /api/profile?handle=deupaxx`
2. Fetches `https://substack.com/@{handle}` with a browser User-Agent
3. Extracts `window._preloads` JSON blob from the HTML
4. Fetches `https://substack.com/api/v1/profile/posts?profile_user_id={id}&limit=50`
5. Returns `ProfileResponse` JSON

- [ ] **Step 1: Create the route file**

```typescript
// src/app/api/profile/route.ts
import { NextRequest, NextResponse } from 'next/server';
import type { ProfileData, PostData, ProfileResponse, ApiError } from '../../../types';

function extractPreloads(html: string): Record<string, unknown> {
  const idx = html.indexOf('JSON.parse("');
  if (idx === -1) throw new Error('Could not find _preloads in page HTML');
  const start = idx + 12;
  const end = html.indexOf('")', start);
  if (end === -1) throw new Error('Could not find end of _preloads JSON');
  const escaped = html.slice(start, end);
  // The HTML contains a JSON string that is itself JSON — parse twice
  return JSON.parse(JSON.parse('"' + escaped + '"'));
}

export async function GET(
  request: NextRequest
): Promise<NextResponse<ProfileResponse | ApiError>> {
  const handle = request.nextUrl.searchParams.get('handle')?.trim();

  if (!handle || !/^[a-zA-Z0-9_-]+$/.test(handle)) {
    return NextResponse.json({ error: 'Invalid handle' }, { status: 400 });
  }

  // 1. Fetch profile page
  let profileRes: Response;
  try {
    profileRes = await fetch(`https://substack.com/@${handle}`, {
      headers: {
        'User-Agent':
          'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/124.0 Safari/537.36',
        Accept: 'text/html',
      },
      next: { revalidate: 0 },
    });
  } catch (e) {
    return NextResponse.json({ error: 'Failed to reach Substack' }, { status: 502 });
  }

  if (!profileRes.ok) {
    return NextResponse.json(
      { error: `Substack returned ${profileRes.status}` },
      { status: 502 }
    );
  }

  const html = await profileRes.text();

  // 2. Parse _preloads
  let preloads: Record<string, unknown>;
  try {
    preloads = extractPreloads(html);
  } catch (e) {
    return NextResponse.json(
      { error: 'Could not parse Substack profile data' },
      { status: 502 }
    );
  }

  const raw = preloads.profile as Record<string, unknown> | null;
  if (!raw) {
    return NextResponse.json({ error: 'Profile not found' }, { status: 404 });
  }

  const userId = raw.id as number;

  const primaryPub = raw.primaryPublication as Record<string, unknown> | null;
  const hasPaidTier = !!(
    primaryPub?.paid_subscription_benefit_title ||
    (raw.max_pub_tier && raw.max_pub_tier !== 'free')
  );

  const profile: ProfileData = {
    id: userId,
    name: raw.name as string,
    handle: raw.handle as string,
    photo_url: (raw.photo_url as string | null) ?? null,
    bio: (raw.bio as string | null) ?? null,
    subscriberCount: (raw.subscriberCount as string | null) ?? null,
    subscriberCountNumber: (raw.subscriberCountNumber as number | null) ?? null,
    followerCount: (raw.followerCount as number) ?? 0,
    hasPaidTier,
  };

  // 3. Fetch posts
  let postsData: { posts: Record<string, unknown>[] };
  try {
    const postsRes = await fetch(
      `https://substack.com/api/v1/profile/posts?profile_user_id=${userId}&limit=50`,
      {
        headers: {
          'User-Agent':
            'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/124.0 Safari/537.36',
        },
        next: { revalidate: 0 },
      }
    );
    if (!postsRes.ok) throw new Error(`Posts API returned ${postsRes.status}`);
    postsData = await postsRes.json();
  } catch (e) {
    return NextResponse.json({ error: 'Failed to fetch posts' }, { status: 502 });
  }

  // 4. Map and sort posts
  const posts: PostData[] = (postsData.posts ?? [])
    .map((p: Record<string, unknown>) => ({
      title: (p.title as string) ?? '(untitled)',
      subtitle: (p.subtitle as string | null) ?? null,
      canonical_url: p.canonical_url as string,
      heartCount:
        ((p.reactions as Record<string, number> | null)?.['❤'] ?? 0),
      restacks: (p.restacks as number) ?? 0,
      audience: (p.audience as string) ?? 'everyone',
      post_date: p.post_date as string,
    }))
    .sort((a, b) => b.heartCount - a.heartCount)
    .slice(0, 10);

  return NextResponse.json({ profile, topPosts: posts });
}
```

// src/app/page.tsx
'use client';

import { useState } from 'react';
import Image from 'next/image';
import type { ProfileResponse } from '../types';

export default function Home() {
  const [handle, setHandle] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [data, setData] = useState<ProfileResponse | null>(null);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    const h = handle.trim().replace(/^@/, '');
    if (!h) return;
    setLoading(true);
    setError(null);
    setData(null);
    try {
      const res = await fetch(`/api/profile?handle=${encodeURIComponent(h)}`);
      const json = await res.json();
      if (!res.ok) {
        setError(json.error ?? 'Something went wrong');
      } else {
        setData(json as ProfileResponse);
      }
    } catch {
      setError('Network error — could not reach server');
    } finally {
      setLoading(false);
    }
  }

  return (
    <main className="min-h-screen bg-gray-50 py-12 px-4">
      <div className="max-w-2xl mx-auto">
        <h1 className="text-3xl font-bold text-gray-900 mb-2">SubScope</h1>
        <p className="text-gray-500 mb-8">Substack profile analytics</p>

        {/* Search form */}
        <form onSubmit={handleSubmit} className="flex gap-2 mb-8">
          <input
            type="text"
            value={handle}
            onChange={(e) => setHandle(e.target.value)}
            placeholder="Substack handle (e.g. deupaxx)"
            className="flex-1 border border-gray-300 rounded px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
          />
          <button
            type="submit"
            disabled={loading}
            className="bg-blue-600 text-white px-4 py-2 rounded text-sm font-medium disabled:opacity-50 hover:bg-blue-700"
          >
            {loading ? 'Loading…' : 'Analyze'}
          </button>
        </form>

        {/* Error */}
        {error && (
          <div className="bg-red-50 border border-red-200 text-red-700 rounded px-4 py-3 mb-6 text-sm">
            {error}
          </div>
        )}

        {/* Profile card */}
        {data && (
          <>
            <div className="bg-white border border-gray-200 rounded-lg p-6 mb-6">
              <div className="flex items-center gap-4 mb-4">
                {data.profile.photo_url ? (
                  <Image
                    src={data.profile.photo_url}
                    alt={data.profile.name}
                    width={64}
                    height={64}
                    className="rounded-full object-cover"
                    unoptimized
                  />
                ) : (
                  <div className="w-16 h-16 rounded-full bg-gray-200 flex items-center justify-center text-gray-400 text-2xl font-bold">
                    {data.profile.name[0]}
                  </div>
                )}
                <div>
                  <h2 className="text-xl font-semibold text-gray-900">{data.profile.name}</h2>
                  <p className="text-sm text-gray-500">@{data.profile.handle}</p>
                </div>
                <div className="ml-auto">
                  <span
                    className={`text-xs font-medium px-2 py-1 rounded-full ${
                      data.profile.hasPaidTier
                        ? 'bg-amber-100 text-amber-800'
                        : 'bg-gray-100 text-gray-600'
                    }`}
                  >
                    {data.profile.hasPaidTier ? 'Paid' : 'Free'}
                  </span>
                </div>
              </div>

              {data.profile.bio && (
                <p className="text-sm text-gray-600 mb-4">{data.profile.bio}</p>
              )}

              <div className="flex gap-6 text-sm">
                {data.profile.subscriberCount && (
                  <div>
                    <span className="font-semibold text-gray-900">
                      {data.profile.subscriberCount}
                    </span>
                    <span className="text-gray-500 ml-1">subscribers</span>
                  </div>
                )}
                <div>
                  <span className="font-semibold text-gray-900">
                    {data.profile.followerCount.toLocaleString()}
                  </span>
                  <span className="text-gray-500 ml-1">followers</span>
                </div>
              </div>
            </div>

            {/* Top posts */}
            <h3 className="text-lg font-semibold text-gray-900 mb-3">
              Top Posts by Hearts
            </h3>
            {data.topPosts.length === 0 ? (
              <p className="text-gray-500 text-sm">No posts found.</p>
            ) : (
              <ol className="space-y-3">
                {data.topPosts.map((post, i) => (
                  <li
                    key={post.canonical_url}
                    className="bg-white border border-gray-200 rounded-lg px-4 py-4 flex gap-3 items-start"
                  >
                    <span className="text-gray-400 text-sm font-mono w-6 pt-0.5 shrink-0">
                      {i + 1}
                    </span>
                    <div className="flex-1 min-w-0">
                      <a
                        href={post.canonical_url}
                        target="_blank"
                        rel="noopener noreferrer"
                        className="text-sm font-medium text-gray-900 hover:text-blue-600 line-clamp-2"
                      >
                        {post.title}
                      </a>
                      {post.subtitle && (
                        <p className="text-xs text-gray-500 mt-0.5 line-clamp-1">
                          {post.subtitle}
                        </p>
                      )}
                      <div className="flex items-center gap-3 mt-2 text-xs text-gray-400">
                        <span>❤ {post.heartCount}</span>
                        <span>🔁 {post.restacks}</span>
                        {post.audience === 'only_paid' && (
                          <span className="text-amber-600 font-medium">Paid</span>
                        )}
                      </div>
                    </div>
                  </li>
                ))}
              </ol>
            )}
          </>
        )}
      </div>
    </main>
  );
}
